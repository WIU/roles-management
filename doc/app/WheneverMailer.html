def adsync_report(email, log)
  @log = log
  logger.info "Sending AD Sync report e-mail to #{email}..."
  mail(:to => email, :subject => "DSS-RM: AD Sync Report for %s" % [Time.now.strftime("%A %B %d, %Y")])
end
